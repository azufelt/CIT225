-- -----------------------------------------------------

CREATE SCHEMA IF NOT EXISTS `mydb` DEFAULT CHARACTER SET utf8 ;
USE `mydb` ;

-- -----------------------------------------------------
-- Table `mydb`.`USERS`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`USERS` (
)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `mydb`.`CIRCLES_GROUP`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`CIRCLES_GROUP` (
  `idCIRCLES_GROUP` INT NOT NULL AUTO_INCREMENT,
  `calling_name` VARCHAR(45) NULL,
  `ORGANIZING_CIRCLE_idORGANIZING_CIRCLE` INT NOT NULL,
  PRIMARY KEY (`idCIRCLES_GROUP`, `ORGANIZING_CIRCLE_idORGANIZING_CIRCLE`),
  UNIQUE INDEX `idCIRCLES_GROUP_UNIQUE` (`idCIRCLES_GROUP` ASC) VISIBLE,
  UNIQUE INDEX `calling_name_UNIQUE` (`calling_name` ASC) VISIBLE)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `mydb`.`USER`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`USER` (
  `idUSER` INT NOT NULL AUTO_INCREMENT,
  `username` VARCHAR(45) NOT NULL,
  `password` VARCHAR(45) NOT NULL,
  `email` VARCHAR(45) NOT NULL,
  `first_name` VARCHAR(45) NOT NULL,
  `last_name` VARCHAR(45) NOT NULL,
  `phone_num` INT NOT NULL,
  `CIRCLES_GROUP_idCIRCLES_GROUP` INT NOT NULL,
  `CIRCLES_GROUP_ORGANIZING_CIRCLE_idORGANIZING_CIRCLE` INT NOT NULL,
  PRIMARY KEY (`idUSER`, `CIRCLES_GROUP_idCIRCLES_GROUP`, `CIRCLES_GROUP_ORGANIZING_CIRCLE_idORGANIZING_CIRCLE`),
  UNIQUE INDEX `idUSER_UNIQUE` (`idUSER` ASC) VISIBLE,
  UNIQUE INDEX `username_UNIQUE` (`username` ASC) VISIBLE,
  UNIQUE INDEX `password_UNIQUE` (`password` ASC) VISIBLE,
  INDEX `fk_USER_CIRCLES_GROUP1_idx` (`CIRCLES_GROUP_idCIRCLES_GROUP` ASC, `CIRCLES_GROUP_ORGANIZING_CIRCLE_idORGANIZING_CIRCLE` ASC) VISIBLE,
  CONSTRAINT `fk_USER_CIRCLES_GROUP1`
    FOREIGN KEY (`CIRCLES_GROUP_idCIRCLES_GROUP` , `CIRCLES_GROUP_ORGANIZING_CIRCLE_idORGANIZING_CIRCLE`)
    REFERENCES `mydb`.`CIRCLES_GROUP` (`idCIRCLES_GROUP` , `ORGANIZING_CIRCLE_idORGANIZING_CIRCLE`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `mydb`.`CALENDAR`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`CALENDAR` (
  `idCALENDAR` INT NOT NULL AUTO_INCREMENT,
  PRIMARY KEY (`idCALENDAR`),
  UNIQUE INDEX `idCALENDAR_UNIQUE` (`idCALENDAR` ASC) VISIBLE)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `mydb`.`DATE-MODULE`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`DATE-MODULE` (
  `idDATE-MODULE` INT NOT NULL AUTO_INCREMENT,
  `dayofyear` VARCHAR(45) NULL,
  `day` VARCHAR(45) NULL,
  `week` VARCHAR(45) NULL,
  `month` VARCHAR(45) NULL,
  `year` VARCHAR(45) NULL,
  `CALENDAR_idCALENDAR` INT NOT NULL,
  PRIMARY KEY (`idDATE-MODULE`, `CALENDAR_idCALENDAR`),
  UNIQUE INDEX `idDATE-MODULE_UNIQUE` (`idDATE-MODULE` ASC) VISIBLE,
  INDEX `fk_DATE-MODULE_CALENDAR1_idx` (`CALENDAR_idCALENDAR` ASC) VISIBLE,
  CONSTRAINT `fk_DATE-MODULE_CALENDAR1`
    FOREIGN KEY (`CALENDAR_idCALENDAR`)
    REFERENCES `mydb`.`CALENDAR` (`idCALENDAR`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `mydb`.`PROJECTS`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`PROJECTS` (
  `idPROJECTS` INT NOT NULL AUTO_INCREMENT,
  `title` VARCHAR(45) NULL,
  `location` VARCHAR(45) NULL,
  `description` VARCHAR(45) NULL,
  `start_date` VARCHAR(45) NULL,
  `CALENDAR_idCALENDAR` INT NOT NULL,
  `DATE-MODULE_idDATE-MODULE` INT NOT NULL,
  `DATE-MODULE_CALENDAR_idCALENDAR` INT NOT NULL,
  `CIRCLES_GROUP_idCIRCLES_GROUP` INT NOT NULL,
  `CIRCLES_GROUP_ORGANIZING_CIRCLE_idORGANIZING_CIRCLE` INT NOT NULL,
  PRIMARY KEY (`idPROJECTS`, `DATE-MODULE_idDATE-MODULE`, `DATE-MODULE_CALENDAR_idCALENDAR`),
  UNIQUE INDEX `idPROJECTS_UNIQUE` (`idPROJECTS` ASC) VISIBLE,
  INDEX `fk_PROJECTS_CALENDAR1_idx` (`CALENDAR_idCALENDAR` ASC) VISIBLE,
  INDEX `fk_PROJECTS_DATE-MODULE1_idx` (`DATE-MODULE_idDATE-MODULE` ASC, `DATE-MODULE_CALENDAR_idCALENDAR` ASC) VISIBLE,
  INDEX `fk_PROJECTS_CIRCLES_GROUP1_idx` (`CIRCLES_GROUP_idCIRCLES_GROUP` ASC, `CIRCLES_GROUP_ORGANIZING_CIRCLE_idORGANIZING_CIRCLE` ASC) VISIBLE,
  CONSTRAINT `fk_PROJECTS_CALENDAR1`
    FOREIGN KEY (`CALENDAR_idCALENDAR`)
    REFERENCES `mydb`.`CALENDAR` (`idCALENDAR`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_PROJECTS_DATE-MODULE1`
    FOREIGN KEY (`DATE-MODULE_idDATE-MODULE` , `DATE-MODULE_CALENDAR_idCALENDAR`)
    REFERENCES `mydb`.`DATE-MODULE` (`idDATE-MODULE` , `CALENDAR_idCALENDAR`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_PROJECTS_CIRCLES_GROUP1`
    FOREIGN KEY (`CIRCLES_GROUP_idCIRCLES_GROUP` , `CIRCLES_GROUP_ORGANIZING_CIRCLE_idORGANIZING_CIRCLE`)
    REFERENCES `mydb`.`CIRCLES_GROUP` (`idCIRCLES_GROUP` , `ORGANIZING_CIRCLE_idORGANIZING_CIRCLE`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `mydb`.`TASK-OWNER`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`TASK-OWNER` (
  `idTASK-OWNER` INT NOT NULL AUTO_INCREMENT,
  `USER_idUSER` INT NOT NULL,
  PRIMARY KEY (`idTASK-OWNER`, `USER_idUSER`),
  UNIQUE INDEX `idTASK-OWNER_UNIQUE` (`idTASK-OWNER` ASC) VISIBLE,
  INDEX `fk_TASK-OWNER_USER1_idx` (`USER_idUSER` ASC) VISIBLE,
  CONSTRAINT `fk_TASK-OWNER_USER1`
    FOREIGN KEY (`USER_idUSER`)
    REFERENCES `mydb`.`USER` (`idUSER`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `mydb`.`STATUS_LIST`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`STATUS_LIST` (
  `idSTATUS` INT NOT NULL AUTO_INCREMENT,
  `ideas` VARCHAR(45) NULL,
  `to-do` VARCHAR(45) NULL,
  `in-progress` VARCHAR(45) NULL,
  `done` VARCHAR(45) NULL,
  `needs attention` VARCHAR(45) NULL,
  PRIMARY KEY (`idSTATUS`),
  UNIQUE INDEX `idSTATUS_UNIQUE` (`idSTATUS` ASC) VISIBLE)
ENGINE = InnoDB;


-- -----------------------------------------------------
-- Table `mydb`.`TASKS`
-- -----------------------------------------------------
CREATE TABLE IF NOT EXISTS `mydb`.`TASKS` (
  `idTASKS` INT NOT NULL AUTO_INCREMENT,
  `assign-date` DATETIME NULL,
  `end-date` DATETIME NULL,
  `description` VARCHAR(45) NULL,
  `budget` INT NULL,
  `TASK-OWNER_idTASK-OWNER` INT NOT NULL,
  `STATUS_LIST_idSTATUS` INT NOT NULL,
  `DATE-MODULE_idDATE-MODULE` INT NOT NULL,
  `DATE-MODULE_CALENDAR_idCALENDAR` INT NOT NULL,
  `PROJECTS_idPROJECTS` INT NOT NULL,
  `PROJECTS_DATE-MODULE_idDATE-MODULE` INT NOT NULL,
  `PROJECTS_DATE-MODULE_CALENDAR_idCALENDAR` INT NOT NULL,
  PRIMARY KEY (`idTASKS`, `TASK-OWNER_idTASK-OWNER`, `PROJECTS_idPROJECTS`, `PROJECTS_DATE-MODULE_idDATE-MODULE`, `PROJECTS_DATE-MODULE_CALENDAR_idCALENDAR`),
  UNIQUE INDEX `idTASKS_UNIQUE` (`idTASKS` ASC) VISIBLE,
  INDEX `fk_TASKS_TASK-OWNER1_idx` (`TASK-OWNER_idTASK-OWNER` ASC) VISIBLE,
  INDEX `fk_TASKS_STATUS_LIST1_idx` (`STATUS_LIST_idSTATUS` ASC) VISIBLE,
  INDEX `fk_TASKS_DATE-MODULE1_idx` (`DATE-MODULE_idDATE-MODULE` ASC, `DATE-MODULE_CALENDAR_idCALENDAR` ASC) VISIBLE,
  INDEX `fk_TASKS_PROJECTS1_idx` (`PROJECTS_idPROJECTS` ASC, `PROJECTS_DATE-MODULE_idDATE-MODULE` ASC, `PROJECTS_DATE-MODULE_CALENDAR_idCALENDAR` ASC) VISIBLE,
  CONSTRAINT `fk_TASKS_TASK-OWNER1`
    FOREIGN KEY (`TASK-OWNER_idTASK-OWNER`)
    REFERENCES `mydb`.`TASK-OWNER` (`idTASK-OWNER`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_TASKS_STATUS_LIST1`
    FOREIGN KEY (`STATUS_LIST_idSTATUS`)
    REFERENCES `mydb`.`STATUS_LIST` (`idSTATUS`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_TASKS_DATE-MODULE1`
    FOREIGN KEY (`DATE-MODULE_idDATE-MODULE` , `DATE-MODULE_CALENDAR_idCALENDAR`)
    REFERENCES `mydb`.`DATE-MODULE` (`idDATE-MODULE` , `CALENDAR_idCALENDAR`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_TASKS_PROJECTS1`
    FOREIGN KEY (`PROJECTS_idPROJECTS` , `PROJECTS_DATE-MODULE_idDATE-MODULE` , `PROJECTS_DATE-MODULE_CALENDAR_idCALENDAR`)
    REFERENCES `mydb`.`PROJECTS` (`idPROJECTS` , `DATE-MODULE_idDATE-MODULE` , `DATE-MODULE_CALENDAR_idCALENDAR`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;
